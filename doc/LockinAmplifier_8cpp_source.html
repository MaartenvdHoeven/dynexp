25  void LockinAmplifierDefs::LockinSample::UpdatePolar() noexcept
26  {
27  PolarResult.R = std::sqrt(CartesianResult.X * CartesianResult.X + CartesianResult.Y * CartesianResult.Y);
28  PolarResult.Theta = std::atan(CartesianResult.Y / CartesianResult.X);
29  }

32  {
